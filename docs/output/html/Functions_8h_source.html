159void readSlots(studentSet* students, classSet* classes, courseSet* courses, slotSet* slots) {
160 ifstream slotsFile(classesSlots);
161 string line;
162 getline(slotsFile, line); // skip first line
163 while (getline(slotsFile, line)) {
164 stringstream ss(line);
165 string classcode, coursecode, day, type;
166 float start, end;
167 getline(ss, classcode, ',');
168 getline(ss, coursecode, ',');
169 getline(ss, day, ',');
170 ss >> start;
171 ss.ignore();
172 ss >> end; end += start;
173 ss.ignore();
174 ss >> type;
175 auto slot = new Slot(day, start, end, type, coursecode, classcode);
176 slots->insert(slot);
177
178 auto class_it = classes->find(new Class(classcode));
179 if (class_it != classes->end()) {
180 (*class_it)->addSlot(*slot);
181 }
182
183 auto course_it = courses->find(new Course(coursecode));
184 if (course_it != courses->end()) {
185 (*course_it)->addSlot(*slot);
186 }
187 }
188 cout << "Read Slots Successfully!" << endl;
