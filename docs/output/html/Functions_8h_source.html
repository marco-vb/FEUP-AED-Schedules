158void readSlots(studentSet* students, classSet* classes, courseSet* courses, slotSet* slots) {
159 ifstream slotsFile(classesSlots);
160 string line;
161 getline(slotsFile, line); // skip first line
162 while (getline(slotsFile, line)) {
163 stringstream ss(line);
164 string classcode, coursecode, day, type;
165 float start, end;
166 getline(ss, classcode, ',');
167 getline(ss, coursecode, ',');
168 getline(ss, day, ',');
169 ss >> start;
170 ss.ignore();
171 ss >> end; end += start;
172 ss.ignore();
173 ss >> type;
174 auto slot = new Slot(day, start, end, type, coursecode, classcode);
175 slots->insert(slot);
176
177 auto class_it = classes->find(new Class(classcode));
178 if (class_it != classes->end()) {
179 (*class_it)->addSlot(*slot);
180 }
181
182 auto course_it = courses->find(new Course(coursecode));
183 if (course_it != courses->end()) {
184 (*course_it)->addSlot(*slot);
185 }
186 }
187 cout << "Read Slots Successfully!" << endl;
188}
