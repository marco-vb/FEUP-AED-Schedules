97void readSlots(studentSet* students, classSet* classes, courseSet* courses, slotSet* slots) {
98 ifstream slotsFile(classesSlots);
99 string line;
100 getline(slotsFile, line); // skip first line
101 while (getline(slotsFile, line)) {
102 stringstream ss(line);
103 string classcode, coursecode, day, type;
104 float start, end;
105 getline(ss, classcode, ',');
106 getline(ss, coursecode, ',');
107 getline(ss, day, ',');
108 ss >> start;
109 ss.ignore();
110 ss >> end; end += start;
111 ss.ignore();
112 ss >> type;
113 auto slot = new Slot(day, start, end, type, coursecode, classcode);
114 slots->insert(slot);
115 auto class_it = classes->find(new Class(classcode));
116 if (class_it != classes->end()) {
117 (*class_it)->addSlot(*slot);
118 }
119
120 auto course_it = courses->find(new Course(coursecode));
121 if (course_it != courses->end()) {
122 (*course_it)->addSlot(*slot);
123 }
124 }
