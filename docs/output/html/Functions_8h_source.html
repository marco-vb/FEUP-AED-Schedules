150void readSlots(studentSet* students, classSet* classes, courseSet* courses, slotSet* slots) {
151 ifstream slotsFile(classesSlots);
152 string line;
153 getline(slotsFile, line); // skip first line
154 while (getline(slotsFile, line)) {
155 stringstream ss(line);
156 string classcode, coursecode, day, type;
157 float start, end;
158 getline(ss, classcode, ',');
159 getline(ss, coursecode, ',');
160 getline(ss, day, ',');
161 ss >> start;
162 ss.ignore();
163 ss >> end; end += start;
164 ss.ignore();
165 ss >> type;
166 auto slot = new Slot(day, start, end, type, coursecode, classcode);
167 slots->insert(slot);
168 auto class_it = classes->find(new Class(classcode));
169 if (class_it != classes->end()) {
170 (*class_it)->addSlot(*slot);
171 }
172
173 auto course_it = courses->find(new Course(coursecode));
174 if (course_it != courses->end()) {
175 (*course_it)->addSlot(*slot);
176 }
177 }
178 cout << "Read Slots Successfully!" << endl;
179}
